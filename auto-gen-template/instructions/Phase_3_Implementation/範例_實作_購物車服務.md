# 實作：購物車服務 範例

展示購物車服務核心邏輯與公開介面範例。

```typescript
export function calculateTotal(items: {price:number, qty:number}[], opts={}) {
  const subtotal = items.reduce((s,i) => s + i.price * i.qty, 0);
  const discount = opts.discount ? subtotal * opts.discount : 0;
  const tax = opts.tax ? (subtotal - discount) * opts.tax : 0;
  return subtotal - discount + tax;
}
```
